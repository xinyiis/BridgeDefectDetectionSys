## 1. 模块概述

### 1.1 功能定位

缺陷检测模块是桥梁缺陷检测系统的**核心业务模块**，负责：
- 图片上传和文件管理
- 调用Python AI服务进行缺陷检测（Mock实现）
- 检测结果的保存和管理
- 缺陷记录的CRUD操作
- 跨表权限验证

### 1.2 核心功能

#### 阶段1：图片上传与缺陷管理（本次实现）

1. **图片上传检测**
   - 上传图片（multipart/form-data）
   - 保存到`/uploads/images/`
   - 调用Python服务检测（Mock）
   - 保存结果图到`/uploads/results/`
   - 保存多条缺陷记录到数据库

2. **缺陷列表查询**
   - 分页查询
   - 多条件过滤（桥梁、类型、时间范围）
   - 权限自动过滤（JOIN bridges表）

3. **缺陷详情查看**
   - 查看单条缺陷详情
   - 包含原图、结果图、检测数据
   - 跨表权限验证

4. **缺陷删除**
   - 软删除缺陷记录
   - 异步删除文件（原图+结果图）
   - 跨表权限验证

#### 阶段2：视频流检测（后续实现）

- 实时视频流检测
- WebSocket通信
- 多帧结果聚合

### 1.3 权限规则

| 角色 | 上传检测 | 查看缺陷 | 删除缺陷 |
|------|---------|---------|---------|
| 普通用户 | 自己的桥梁 | 仅自己桥梁的 | 仅自己桥梁的 |
| 管理员 | 所有桥梁 | 全部 | 全部 |

### 2.2 复杂度来源

#### 简化点（相比桥梁模块）

1. **无唯一约束检查**
   - 缺陷记录无需唯一性验证
   - 无需修改编号释放索引

2. **无复杂级联删除**
   - 无子记录关联
   - 只需清理文件

3. **模型字段较少**
   - 13个字段 vs 桥梁18个

#### 复杂点（相比无人机模块）

1. **跨表权限验证**
   - 需要JOIN bridges表验证所有权
   - Service层需要复杂SQL
   - 中间件需要查询bridge再验证

2. **Python服务集成**
   - 需要定义服务接口
   - Mock实现（模拟AI检测）
   - HTTP实现（预留）
   - 错误处理和超时

3. **双文件上传**
   - 原始图片上传
   - 结果图片保存
   - 文件回滚机制

4. **多缺陷处理**
   - 1张图片 → N条缺陷记录
   - 共享image_path和result_path
   - 循环保存逻辑

```go
// List - 支持权限过滤的复杂查询
func (r *DefectRepositoryImpl) List(filters DefectListFilters) ([]model.Defect, int64, error) {
    query := r.db.Model(&model.Defect{})

    // JOIN bridges表用于权限过滤
    query = query.Joins("LEFT JOIN bridges ON bridges.id = defects.bridge_id")

    // 权限过滤（关键）
    if filters.CurrentUser != nil && !filters.CurrentUser.IsAdmin() {
        query = query.Where("bridges.user_id = ?", filters.CurrentUser.ID)
    }

    // 其他过滤条件
    if filters.BridgeID != nil {
        query = query.Where("defects.bridge_id = ?", *filters.BridgeID)
    }
    if filters.DefectType != nil {
        query = query.Where("defects.defect_type = ?", *filters.DefectType)
    }
    // ... 时间过滤

    // 分页
    var total int64
    query.Count(&total)

    var defects []model.Defect
    query.Offset((filters.Page - 1) * filters.PageSize).Limit(filters.PageSize).Find(&defects)

    return defects, total, nil
}
```

```go
package external

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"

)

// HTTPPythonService HTTP实现（调用真实Python服务）
type HTTPPythonService struct {
    BaseURL string
    Client  *http.Client
}

func NewHTTPPythonService(baseURL string) *HTTPPythonService {
    return &HTTPPythonService{
        BaseURL: baseURL,
        Client: &http.Client{
            Timeout: 30 * time.Second,
        },
    }
}

func (s *HTTPPythonService) DetectDefect(imagePath, modelName string, pixelRatio float64) (*service.PythonDetectionResult, error) {
    // 构造请求
    reqData := map[string]interface{}{
        "image_path":  imagePath,
        "model_name":  modelName,
        "pixel_ratio": pixelRatio,
    }

    jsonData, _ := json.Marshal(reqData)

    // 发送HTTP请求
    resp, err := s.Client.Post(
        s.BaseURL+"/api/detect",
        "application/json",
        bytes.NewBuffer(jsonData),
    )
    if err != nil {
        return nil, fmt.Errorf("Python服务请求失败: %w", err)
    }
    defer resp.Body.Close()

    // 解析响应
    body, _ := ioutil.ReadAll(resp.Body)
    var result service.PythonDetectionResult
    if err := json.Unmarshal(body, &result); err != nil {
        return nil, fmt.Errorf("解析Python响应失败: %w", err)
    }

    return &result, nil
}
```

```go
// GetDefect 获取缺陷详情
func (s *DefectService) GetDefect(id uint) (*model.Defect, error) {
    return s.defectRepo.FindByID(id)
}

// DeleteDefect 删除缺陷
func (s *DefectService) DeleteDefect(id uint) error {
    // 软删除
    return s.defectRepo.Delete(id)
}

// VerifyDefectOwnership 验证缺陷所有权（用于中间件）
func (s *DefectService) VerifyDefectOwnership(defectID uint, userID uint, isAdmin bool) (*model.Defect, error) {
    // 1. 查询缺陷
    defect, err := s.defectRepo.FindByID(defectID)
    if err != nil {
        return nil, err
    }

    // 2. 管理员直接放行
    if isAdmin {
        return defect, nil
    }

    // 3. 查询桥梁
    bridge, err := s.bridgeRepo.FindByID(defect.BridgeID)
    if err != nil {
        return nil, errors.New("关联桥梁不存在")
    }

    // 4. 验证所有权
    if bridge.UserID != userID {
        return nil, errors.New("无权访问此缺陷")
    }

    return defect, nil
}
```

**关键点**：
- CreateDefect验证桥梁存在性
- ListDefects传递CurrentUser给Repository
- VerifyDefectOwnership用于中间件跨表验证

```go
type DetectionUseCase struct {
    defectService  *service.DefectService
    bridgeService  *service.BridgeService
    pythonService  service.PythonService

}

// UploadAndDetect 上传图片并检测
func (uc *DetectionUseCase) UploadAndDetect(req *dto.DetectionUploadRequest, currentUser *model.User) (*dto.DetectionResponse, error) {
    startTime := time.Now()

    // 1. 验证桥梁权限
    bridge, err := uc.bridgeService.GetBridge(req.BridgeID, currentUser)
    if err != nil {
        return nil, errors.New("桥梁不存在或无权访问")
    }

    // 2. 保存原始图片
    imagePath, err := uc.fileService.SaveImage(req.Image, "images")
    if err != nil {
        return nil, fmt.Errorf("图片保存失败: %w", err)
    }

    // 3. 调用Python服务检测
    pythonResult, err := uc.pythonService.DetectDefect(imagePath, req.ModelName, req.PixelRatio)
    if err != nil {
        // 回滚：删除已上传图片
        uc.fileService.DeleteFile(imagePath)
        return nil, fmt.Errorf("AI检测失败: %w", err)
    }

    // 4. 保存结果图
    var resultPath string
    if pythonResult.ResultImage != "" {
        resultPath, err = uc.fileService.SaveResultImage(pythonResult.ResultImage, "results")
        if err != nil {
            // 回滚：删除原图
            uc.fileService.DeleteFile(imagePath)
            return nil, fmt.Errorf("结果图保存失败: %w", err)
        }
    }

    // 5. 保存每个检测到的缺陷
    defects := make([]*model.Defect, 0, len(pythonResult.Defects))
    for _, detectedDefect := range pythonResult.Defects {
        defect := &model.Defect{
            BridgeID:   req.BridgeID,
            DefectType: detectedDefect.DefectType,
            ImagePath:  imagePath,      // 共享
            ResultPath: resultPath,     // 共享
            BBox:       detectedDefect.BBoxJSON(),
            Length:     detectedDefect.Length,
            Width:      detectedDefect.Width,
            Area:       detectedDefect.Area,
            Confidence: detectedDefect.Confidence,
            DetectedAt: time.Now(),
        }

        if err := uc.defectService.CreateDefect(defect); err != nil {
            // 回滚：删除已上传文件
            uc.fileService.DeleteFile(imagePath)
            uc.fileService.DeleteFile(resultPath)
            return nil, fmt.Errorf("缺陷保存失败: %w", err)
        }

        defects = append(defects, defect)
    }

    // 6. 构造响应
    return &dto.DetectionResponse{
        TotalDefects:   len(defects),
        ImagePath:      imagePath,
        ResultPath:     resultPath,
        ProcessingTime: time.Since(startTime).Seconds(),
        Defects:        uc.toDefectDTOs(defects),
        DefectSummary:  uc.buildDefectSummary(defects),
    }, nil
}

// toDefectDTOs DTO转换
func (uc *DetectionUseCase) toDefectDTOs(defects []*model.Defect) []dto.DefectDTO {
    dtos := make([]dto.DefectDTO, len(defects))
    for i, d := range defects {
        dtos[i] = dto.DefectDTO{
            ID:         d.ID,
            BridgeID:   d.BridgeID,
            DefectType: d.DefectType,
            ImagePath:  d.ImagePath,
            ResultPath: d.ResultPath,
            BBox:       d.BBox,
            Length:     d.Length,
            Width:      d.Width,
            Area:       d.Area,
            Confidence: d.Confidence,
            DetectedAt: d.DetectedAt,
        }
    }
    return dtos
}

// buildDefectSummary 构建缺陷统计
func (uc *DetectionUseCase) buildDefectSummary(defects []*model.Defect) map[string]int {
    summary := make(map[string]int)
    for _, d := range defects {
        summary[d.DefectType]++
    }
    return summary
}
```

```go
// DefectOwnershipRequired 缺陷所有权验证中间件
func DefectOwnershipRequired(defectService *service.DefectService) gin.HandlerFunc {
    return func(c *gin.Context) {
        // 1. 获取当前用户
        currentUser := c.MustGet("current_user").(*model.User)

        // 2. 管理员直接放行
        if currentUser.IsAdmin() {
            c.Next()
            return
        }

        // 3. 获取缺陷ID
        defectIDStr := c.Param("id")
        defectID, err := strconv.ParseUint(defectIDStr, 10, 64)
        if err != nil {
            response.BadRequest(c, "无效的缺陷ID")
            c.Abort()
            return
        }

        // 4. 验证所有权（跨表查询）
        defect, err := defectService.VerifyDefectOwnership(uint(defectID), currentUser.ID, currentUser.IsAdmin())
        if err != nil {
            response.Forbidden(c, "无权访问此缺陷")
            c.Abort()
            return
        }

        // 5. 存入上下文
        c.Set("defect", defect)
        c.Next()
    }
}
```

**关键点**：
- 调用DefectService.VerifyDefectOwnership进行跨表验证
- 管理员bypass优化
- Context缓存避免重复查询

## 9. 设计要点总结

### 9.1 与桥梁模块的差异

| 特性 | 桥梁模块 | 缺陷检测模块 |
|------|---------|-------------|
| **唯一约束** | ✅ bridge_code | ❌ 无 |
| **权限验证** | 直接关联user_id | 跨表（JOIN bridges） |
| **文件上传** | 1个（3D模型） | 2个（原图+结果图） |
| **外部服务** | ❌ 无 | ✅ Python服务 |
| **级联删除** | ✅ 删除缺陷 | ✅ 删除文件 |

### 9.2 与无人机模块的差异

| 特性 | 无人机模块 | 缺陷检测模块 |
|------|-----------|-------------|
| **权限验证** | 直接关联user_id | 跨表（JOIN bridges） |
| **删除方式** | 物理删除 | 软删除 |
| **文件处理** | ❌ 无 | ✅ 双文件 |
| **外部服务** | ❌ 无 | ✅ Python服务 |

### 9.3 核心技术

1. **Python服务抽象**：接口定义 + Mock/HTTP双实现
2. **跨表权限验证**：Service层JOIN + 中间件验证
3. **文件上传回滚**：每步检查错误，失败时删除已上传文件
4. **多缺陷处理**：循环保存，共享图片路径
5. **异步文件删除**：goroutine，不阻塞响应
