// Package persistence 提供数据持久化实现
package persistence

import (
    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/model"
    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/repository"
    "gorm.io/gorm"
)

// UserRepository 用户仓储实现
type UserRepository struct {
    db *gorm.DB
}

// NewUserRepository 创建用户仓储实例
func NewUserRepository(db *gorm.DB) repository.IUserRepository {
    return &UserRepository{db: db}
}

// Create 创建用户
func (r *UserRepository) Create(user *model.User) error {
    return r.db.Create(user).Error
}

// FindByID 根据ID查询用户
func (r *UserRepository) FindByID(id uint) (*model.User, error) {
    var user model.User
    err := r.db.First(&user, id).Error
    if err != nil {
        return nil, err
    }
    return &user, nil
}

// FindByUsername 根据用户名查询用户
func (r *UserRepository) FindByUsername(username string) (*model.User, error) {
    var user model.User
    err := r.db.Where("username = ?", username).First(&user).Error
    if err != nil {
        return nil, err
    }
    return &user, nil
}

// FindByEmail 根据邮箱查询用户
func (r *UserRepository) FindByEmail(email string) (*model.User, error) {
    var user model.User
    err := r.db.Where("email = ?", email).First(&user).Error
    if err != nil {
        return nil, err
    }
    return &user, nil
}

// Update 更新用户信息
func (r *UserRepository) Update(user *model.User) error {
    return r.db.Save(user).Error
}

// Delete 删除用户（硬删除）
func (r *UserRepository) Delete(id uint) error {
    return r.db.Delete(&model.User{}, id).Error
}

// List 获取用户列表（分页）
func (r *UserRepository) List(page, pageSize int) ([]model.User, int64, error) {
    var users []model.User
    var total int64

    // 计算总数
    if err := r.db.Model(&model.User{}).Count(&total).Error; err != nil {
        return nil, 0, err
    }

    // 分页查询
    offset := (page - 1) * pageSize
    err := r.db.Offset(offset).Limit(pageSize).Find(&users).Error

    return users, total, err
}

// CountByRole 统计指定角色的用户数量
func (r *UserRepository) CountByRole(role string) (int64, error) {
    var count int64
    err := r.db.Model(&model.User{}).Where("role = ?", role).Count(&count).Error
    return count, err
}

**文件**: `src/backend/internal/domain/service/user_service.go`

```go
// Package service 定义领域服务
package service

import (
    "errors"
    "strings"

    "golang.org/x/crypto/bcrypt"
    "gorm.io/gorm"

    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/model"
    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/repository"
)

// UserService 用户领域服务
type UserService struct {
    userRepo repository.IUserRepository
}

// NewUserService 创建用户服务实例
func NewUserService(userRepo repository.IUserRepository) *UserService {
    return &UserService{
        userRepo: userRepo,
    }
}

// Register 用户注册
func (s *UserService) Register(username, password, realName, phone, email string) (*model.User, error) {
    // 1. 检查用户名是否已存在
    existingUser, err := s.userRepo.FindByUsername(username)
    if err == nil && existingUser != nil {
        return nil, errors.New("用户名已存在")
    }

    // 2. 检查邮箱是否已存在
    existingUser, err = s.userRepo.FindByEmail(email)
    if err == nil && existingUser != nil {
        return nil, errors.New("邮箱已存在")
    }

    // 3. 加密密码
    hashedPassword, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    if err != nil {
        return nil, errors.New("密码加密失败")
    }

    // 4. 创建用户
    user := &model.User{
        Username: username,
        Password: string(hashedPassword),
        RealName: realName,
        Phone:    phone,
        Email:    email,
        Role:     "user", // 默认普通用户
    }

    if err := s.userRepo.Create(user); err != nil {
        return nil, err
    }

    return user, nil
}

// Login 用户登录
func (s *UserService) Login(usernameOrEmail, password string) (*model.User, error) {
    var user *model.User
    var err error

    // 1. 判断是用户名还是邮箱登录
    if strings.Contains(usernameOrEmail, "@") {
        user, err = s.userRepo.FindByEmail(usernameOrEmail)
    } else {
        user, err = s.userRepo.FindByUsername(usernameOrEmail)
    }

    if err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("用户名或密码错误")
        }
        return nil, err
    }

    // 2. 验证密码
    err = bcrypt.CompareHashAndPassword([]byte(user.Password), []byte(password))
    if err != nil {
        return nil, errors.New("用户名或密码错误")
    }

    return user, nil
}

// GetByID 根据ID获取用户
func (s *UserService) GetByID(id uint) (*model.User, error) {
    user, err := s.userRepo.FindByID(id)
    if err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            return nil, errors.New("用户不存在")
        }
        return nil, err
    }
    return user, nil
}

// UpdateUser 更新用户信息
func (s *UserService) UpdateUser(id uint, realName, phone, email, password string) (*model.User, error) {
    // 1. 查询用户
    user, err := s.userRepo.FindByID(id)
    if err != nil {
        return nil, errors.New("用户不存在")
    }

    // 2. 更新字段
    if realName != "" {
        user.RealName = realName
    }
    if phone != "" {
        user.Phone = phone
    }
    if email != "" {
        // 检查邮箱是否被其他用户使用
        existingUser, _ := s.userRepo.FindByEmail(email)
        if existingUser != nil && existingUser.ID != id {
            return nil, errors.New("邮箱已被使用")
        }
        user.Email = email
    }
    if password != "" {
        hashedPassword, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
        if err != nil {
            return nil, errors.New("密码加密失败")
        }
        user.Password = string(hashedPassword)
    }

    // 3. 保存更新
    if err := s.userRepo.Update(user); err != nil {
        return nil, err
    }

    return user, nil
}

// DeleteUser 删除用户
func (s *UserService) DeleteUser(id uint) error {
    return s.userRepo.Delete(id)
}

// PromoteUser 提升用户为管理员
func (s *UserService) PromoteUser(id uint) (*model.User, error) {
    user, err := s.userRepo.FindByID(id)
    if err != nil {
        return nil, errors.New("用户不存在")
    }

    if user.IsAdmin() {
        return nil, errors.New("用户已经是管理员")
    }

    user.Role = "admin"
    if err := s.userRepo.Update(user); err != nil {
        return nil, err
    }

    return user, nil
}

// GetUserList 获取用户列表
func (s *UserService) GetUserList(page, pageSize int) ([]model.User, int64, error) {
    return s.userRepo.List(page, pageSize)
}
```

// Package handler 定义HTTP处理器
package handler

import (
    "strconv"

    "github.com/gin-contrib/sessions"
    "github.com/gin-gonic/gin"

    "github.com/xinyiis/BridgeDefectDetectionSys/internal/application/dto"
    "github.com/xinyiis/BridgeDefectDetectionSys/internal/application/usecase"
    "github.com/xinyiis/BridgeDefectDetectionSys/pkg/response"
)

// UserHandler 用户处理器
type UserHandler struct {
    registerUseCase *usecase.RegisterUseCase
    loginUseCase    *usecase.LoginUseCase
    userMgmtUseCase *usecase.UserManagementUseCase
}

// NewUserHandler 创建用户处理器
func NewUserHandler(
    registerUseCase *usecase.RegisterUseCase,
    loginUseCase *usecase.LoginUseCase,
    userMgmtUseCase *usecase.UserManagementUseCase,
) *UserHandler {
    return &UserHandler{
        registerUseCase: registerUseCase,
        loginUseCase:    loginUseCase,
        userMgmtUseCase: userMgmtUseCase,
    }
}

// Register 用户注册
func (h *UserHandler) Register(c *gin.Context) {
    var req dto.RegisterRequest

    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "参数格式错误: "+err.Error())
        return
    }

    user, err := h.registerUseCase.Execute(req)
    if err != nil {
        if err.Error() == "用户名已存在" || err.Error() == "邮箱已存在" {
            response.Conflict(c, err.Error())
            return
        }
        response.InternalError(c)
        return
    }

    response.Success(c, gin.H{
        "user_id":  user.ID,
        "username": user.Username,
        "role":     user.Role,
    })
}

// Login 用户登录
func (h *UserHandler) Login(c *gin.Context) {
    var req dto.LoginRequest

    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "参数格式错误")
        return
    }

    user, err := h.loginUseCase.Execute(req)
    if err != nil {
        response.Unauthorized(c)
        return
    }

    // 创建Session
    session := sessions.Default(c)
    session.Set("user_id", user.ID)
    if err := session.Save(); err != nil {
        response.InternalError(c)
        return
    }

    response.Success(c, gin.H{
        "user": dto.UserResponse{
            UserID:    user.ID,
            Username:  user.Username,
            RealName:  user.RealName,
            Email:     user.Email,
            Role:      user.Role,
            CreatedAt: user.CreatedAt.Format("2006-01-02T15:04:05Z07:00"),
        },
    })
}

// Logout 用户登出
func (h *UserHandler) Logout(c *gin.Context) {
    session := sessions.Default(c)
    session.Clear()
    session.Save()

    response.Success(c, nil)
}

// GetCurrentUser 获取当前用户信息
func (h *UserHandler) GetCurrentUser(c *gin.Context) {
    user := c.MustGet("current_user").(*model.User)

    response.Success(c, dto.UserResponse{
        UserID:    user.ID,
        Username:  user.Username,
        RealName:  user.RealName,
        Phone:     user.Phone,
        Email:     user.Email,
        Role:      user.Role,
        CreatedAt: user.CreatedAt.Format("2006-01-02T15:04:05Z07:00"),
    })
}

// GetUserList 获取用户列表（管理员）
func (h *UserHandler) GetUserList(c *gin.Context) {
    page, _ := strconv.Atoi(c.DefaultQuery("page", "1"))
    pageSize, _ := strconv.Atoi(c.DefaultQuery("page_size", "10"))

    result, err := h.userMgmtUseCase.GetUserList(page, pageSize)
    if err != nil {
        response.InternalError(c)
        return
    }

    response.Success(c, result)
}

// GetUserByID 获取用户详情
func (h *UserHandler) GetUserByID(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        response.BadRequest(c, "无效的用户ID")
        return
    }

    // 权限检查：只能查看自己或管理员可以查看所有
    currentUser := c.MustGet("current_user").(*model.User)
    if !currentUser.IsAdmin() && currentUser.ID != uint(id) {
        response.Forbidden(c)
        return
    }

    user, err := h.userMgmtUseCase.GetUserByID(uint(id))
    if err != nil {
        response.NotFound(c, "用户不存在")
        return
    }

    response.Success(c, dto.UserResponse{
        UserID:    user.ID,
        Username:  user.Username,
        RealName:  user.RealName,
        Phone:     user.Phone,
        Email:     user.Email,
        Role:      user.Role,
        CreatedAt: user.CreatedAt.Format("2006-01-02T15:04:05Z07:00"),
    })
}

// UpdateUser 更新用户信息
func (h *UserHandler) UpdateUser(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        response.BadRequest(c, "无效的用户ID")
        return
    }

    // 权限检查
    currentUser := c.MustGet("current_user").(*model.User)
    if !currentUser.IsAdmin() && currentUser.ID != uint(id) {
        response.Forbidden(c)
        return
    }

    var req dto.UpdateUserRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, "参数格式错误")
        return
    }

    user, err := h.userMgmtUseCase.UpdateUser(uint(id), req)
    if err != nil {
        if err.Error() == "邮箱已被使用" {
            response.Conflict(c, err.Error())
            return
        }
        response.InternalError(c)
        return
    }

    response.Success(c, gin.H{
        "user_id":  user.ID,
        "username": user.Username,
        "real_name": user.RealName,
        "email":    user.Email,
    })
}

// DeleteUser 删除用户（管理员）
func (h *UserHandler) DeleteUser(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        response.BadRequest(c, "无效的用户ID")
        return
    }

    // 不能删除自己
    currentUser := c.MustGet("current_user").(*model.User)
    if currentUser.ID == uint(id) {
        response.BadRequest(c, "不能删除自己")
        return
    }

    if err := h.userMgmtUseCase.DeleteUser(uint(id)); err != nil {
        response.InternalError(c)
        return
    }

    response.Success(c, nil)
}

// PromoteUser 提升用户权限（管理员）
func (h *UserHandler) PromoteUser(c *gin.Context) {
    id, err := strconv.ParseUint(c.Param("id"), 10, 32)
    if err != nil {
        response.BadRequest(c, "无效的用户ID")
        return
    }

    user, err := h.userMgmtUseCase.PromoteUser(uint(id))
    if err != nil {
        if err.Error() == "用户已经是管理员" {
            response.BadRequest(c, err.Error())
            return
        }
        response.InternalError(c)
        return
    }

    response.Success(c, gin.H{
        "user_id":  user.ID,
        "username": user.Username,
        "role":     user.Role,
    })
}

## Step 6: 实现认证中间件

### 6.1 认证中间件

**文件**: `src/backend/internal/interfaces/http/middleware/auth.go`

```go
// Package middleware 定义HTTP中间件
package middleware

import (
    "github.com/gin-contrib/sessions"
    "github.com/gin-gonic/gin"

    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/service"
    "github.com/xinyiis/BridgeDefectDetectionSys/pkg/response"
)

// AuthRequired 认证中间件
func AuthRequired(userService *service.UserService) gin.HandlerFunc {
    return func(c *gin.Context) {
        session := sessions.Default(c)
        userID := session.Get("user_id")

        if userID == nil {
            response.Unauthorized(c)
            c.Abort()
            return
        }

        // 查询用户信息
        user, err := userService.GetByID(userID.(uint))
        if err != nil {
            response.Unauthorized(c)
            c.Abort()
            return
        }

        // 将用户信息存入上下文
        c.Set("current_user", user)
        c.Next()
    }
}
```

### 6.2 管理员权限中间件

**文件**: `src/backend/internal/interfaces/http/middleware/permission.go`

```go
package middleware

import (
    "github.com/gin-gonic/gin"

    "github.com/xinyiis/BridgeDefectDetectionSys/internal/domain/model"
    "github.com/xinyiis/BridgeDefectDetectionSys/pkg/response"
)

// AdminRequired 管理员权限中间件
func AdminRequired() gin.HandlerFunc {
    return func(c *gin.Context) {
        user, exists := c.Get("current_user")
        if !exists {
            response.Forbidden(c)
            c.Abort()
            return
        }

        currentUser := user.(*model.User)
        if !currentUser.IsAdmin() {
            response.Forbidden(c)
            c.Abort()
            return
        }

        c.Next()
    }
}

func SetupRouter(db *gorm.DB, cfg *config.Config) *gin.Engine {
    // ... 现有代码 ...

    // 初始化仓储
    userRepo := persistence.NewUserRepository(db)

    // 初始化服务
    userService := service.NewUserService(userRepo)

    // 初始化用例
    registerUseCase := usecase.NewRegisterUseCase(userService)
    loginUseCase := usecase.NewLoginUseCase(userService)
    userMgmtUseCase := usecase.NewUserManagementUseCase(userService)

    // 初始化Handler
    userHandler := handler.NewUserHandler(registerUseCase, loginUseCase, userMgmtUseCase)

    // API v1路由组
    v1 := router.Group("/api/v1")
    {
        // 认证接口（无需登录）
        auth := v1.Group("/auth")
        {
            auth.POST("/register", userHandler.Register)
            auth.POST("/login", userHandler.Login)
            auth.POST("/logout", middleware.AuthRequired(userService), userHandler.Logout)
            auth.GET("/me", middleware.AuthRequired(userService), userHandler.GetCurrentUser)
        }

        // 用户管理接口（需要登录）
        users := v1.Group("/users")
        users.Use(middleware.AuthRequired(userService))
        {
            // 管理员权限
            users.GET("", middleware.AdminRequired(), userHandler.GetUserList)

            // 用户自己或管理员
            users.GET("/:id", userHandler.GetUserByID)
            users.PUT("/:id", userHandler.UpdateUser)

            // 仅管理员
            users.DELETE("/:id", middleware.AdminRequired(), userHandler.DeleteUser)
            users.POST("/:id/promote", middleware.AdminRequired(), userHandler.PromoteUser)
        }
    }

    return router
}
```

## 常见问题

### 1. 编译错误：找不到包

**解决**: 确保所有import使用完整的模块路径
```go
import "github.com/xinyiis/BridgeDefectDetectionSys/..."
```

运行 `go mod tidy` 更新依赖

### 2. Session无法保存

**检查**:
- CORS配置中 `AllowCredentials: true`
- 前端请求设置 `credentials: 'include'`
- Session Secret配置正确

### 3. 密码验证失败

**检查**:
- 密码是否使用bcrypt加密存储
- CompareHashAndPassword的参数顺序（hash在前，password在后）

### 4. 权限验证不生效

**检查**:
- 中间件顺序：AuthRequired必须在AdminRequired之前
- Context中是否正确存储了用户信息
