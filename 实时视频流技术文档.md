### 1.2 技术栈

| 层级 | 技术栈 | 作用 |
|------|--------|------|

- **资源可控**: 自动监控内存使用，清理僵尸任务
- **双引擎AI**: YOLO快速定位 + LLM智能判断，降低误报率

```typescript
function connectWebSocket() {
  // 构建WebSocket URL
  const wsUrl = 'ws://localhost:5000/ws'

  // 创建WebSocket连接
  ws = new WebSocket(wsUrl)

  // 连接成功回调
  ws.onopen = () => {
    console.log('✅ WebSocket连接成功')
    wsConnected.value = true
    reconnectAttempts = 0
  }

  // 接收消息回调
  ws.onmessage = handleWebSocketMessage

  // 连接关闭回调
  ws.onclose = () => {
    console.log('❌ WebSocket连接关闭')
    wsConnected.value = false
    // 尝试重连
    attemptReconnect()
  }

  // 连接错误回调
  ws.onerror = (error) => {
    console.error('WebSocket错误:', error)
  }
}
```

**关键配置**:
- WebSocket地址: `ws://localhost:5000/ws`
- 自动重连: 连接断开后每5秒尝试重连
- 最大重连次数: 10次

```typescript
  try {
    const message = JSON.parse(event.data)

    switch (message.type) {
      case 'camera_status':
        handleCameraStatus(message.data)
        break

      case 'detection_result':
        handleDetectionResult(message.data)
        break

      case 'stream_stopped':
        handleStreamStopped(message.data)
        break

      default:
        console.warn('未知消息类型:', message.type)
    }
  } catch (error) {

}
```

### 4.3 WebSocket连接处理

```python
@sock.route('/ws')
def handle_websocket(ws):
    """处理WebSocket连接"""
    # 1. 为连接分配唯一ID
    ws_id = str(id(ws))

    # 2. 注册连接
    with stream_tasks_lock:
        ws_tasks[ws_id] = set()

    print(f"✅ 新的WebSocket连接: {ws_id}")

    try:
        # 3. 消息接收循环
        while True:
            # 接收客户端消息
            data = ws.receive()
            message = json.loads(data)

            # 分发处理
            handle_message(ws, ws_id, message)

    except Exception as e:
        print(f"❌ WebSocket错误: {str(e)}")
    finally:
        # 4. 连接关闭，清理资源
        print(f"WebSocket连接关闭，清理资源: {ws_id}")
        cleanup_ws_tasks(ws_id)
```

```python
# video_stream.py
class VideoStream:
    def start(self) -> bool:
        """启动视频流"""
        # Step 1: 检查RTSP地址可达性
        if not self.check_rtsp_connection():
            print(f"❌ RTSP地址不可达: {self.rtsp_url}")
            return False

        # Step 2: 打开视频流
        try:
            self.cap = cv2.VideoCapture(self.rtsp_url)

            if not self.cap.isOpened():
                print(f"❌ 无法打开视频流: {self.rtsp_url}")
                return False

            # Step 3: 读取一帧测试
            ret, frame = self.cap.read()
            if not ret or frame is None:
                print(f"❌ 无法读取视频帧: {self.rtsp_url}")
                self.stop()
                return False

            self.running = True
            print(f"✅ 视频流连接成功: {self.rtsp_url}")
            return True

        except Exception as e:
            print(f"❌ 视频流连接失败: {e}")
            return False

    def check_rtsp_connection(self) -> bool:
        """检查RTSP地址可达性"""
        import socket
        import re

        # 解析RTSP URL获取主机和端口
        match = re.match(r'rtsp://([^:/]+)(?::(\d+))?', self.rtsp_url)
        if not match:
            return False

        host = match.group(1)
        port = int(match.group(2)) if match.group(2) else 554

        # Socket连接测试
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(2.0)  # 2秒超时

        result = sock.connect_ex((host, port))
        sock.close()

        return result == 0
```

**连接优化**:
- ✅ 预检查: Socket测试避免长时间等待
- ✅ 帧测试: 确保流有效才返回成功
- ✅ 超时控制: 2秒超时，快速失败

```python
# websocket_util.py
def process_stream(ws, ws_id, camera_id, stream):
    """视频流处理主循环"""

    print(f"🎬 开始处理视频流: camera_id={camera_id}")

    last_detection_time = time.time()
    frame_count = 0

    try:
        # 主循环: 持续读取和检测
        while camera_id in stream_tasks and \
              stream_tasks[camera_id]['status'] == 'running':

            # ========== Step 1: 读取视频帧 ==========
            ret, frame = stream.read_frame()
            if not ret or frame is None:
                print(f"❌ 读取帧失败: {camera_id}")
                break

            frame_count += 1
            current_time = time.time()

            # 更新最后活动时间
            with stream_tasks_lock:
                if camera_id in stream_tasks:
                    stream_tasks[camera_id]['last_activity'] = current_time

            # ========== Step 2: 按间隔执行检测 ==========
            if (current_time - last_detection_time) >= FRAME_INTERVAL:

                # 2.1 执行AI检测
                result = video_detector.detect_image(frame)
                last_detection_time = current_time

                detected_objects = result.get('detectedObjects', [])

                # 2.2 绘制检测框
                if DRAW_DETECTION_BOX_ON_SERVER:
                    display_frame = draw_detection_boxes(
                        frame,
                        detected_objects,
                        video_detector.get_class_colors
                    )
                else:
                    display_frame = frame

                # 2.3 编码图像为Base64
                _, buffer = cv2.imencode('.jpg', display_frame)
                base64_frame = base64.b64encode(buffer).decode('utf-8')
                result['frame'] = f"data:image/jpeg;base64,{base64_frame}"

                # 2.4 添加图像尺寸
                if display_frame is not None:
                    height, width = display_frame.shape[:2]
                    result['imageWidth'] = width
                    result['imageHeight'] = height

                # 2.5 发送检测结果
                send_detection_result(ws, ws_id, camera_id, result)

                # ========== Step 3: 火灾事件处理 ==========
                if result.get('isFireIncident', False):
                    print(f"🔥 检测到火灾事件: camera_id={camera_id}")

                    # 延时1秒，等待保存检测记录
                    time.sleep(1.0)

                    # 保存检测记录到数据库
                    save_detection_record_to_db(camera_id, result)

            # 释放帧内存
            del frame
            if 'display_frame' in locals():
                del display_frame

        print(f"🎬 视频流处理结束: camera_id={camera_id}")

    except Exception as e:
        print(f"❌ 处理视频流异常: {e}")
    finally:
        # 清理资源
        stop_stream_and_cleanup(ws, ws_id, camera_id)
```

```python
# video_detect.py
def detect_image(self, image: np.ndarray) -> Dict:
    """检测图像中的火灾和烟雾"""

    # 1. 保存图像为临时文件
    temp_filename = os.path.join(TEMP_DIR, f"detect_{uuid.uuid4().hex}.jpg")
    cv2.imwrite(temp_filename, image)

    # 2. YOLO检测
    results = self.model(temp_filename)

    # 3. 解析检测结果
    detected_objects = []
    for result in results:
        boxes = result.boxes
        for box in boxes:
            # 边界框坐标
            x1, y1, x2, y2 = box.xyxy[0].cpu().numpy()

            # 置信度
            conf = float(box.conf[0].cpu().numpy())

            # 类别
            cls = int(box.cls[0].cpu().numpy())
            cls_name = result.names[cls]  # 'fire' 或 'smoke'

            detected_objects.append({
                'bbox': [float(x1), float(y1), float(x2), float(y2)],
                'confidence': conf,
                'class': cls_name
            })

    # 4. 如果检测到对象，进行LLM分析
    if detected_objects:
        return self._analyze_with_llm(image, detected_objects)
    else:
        return {
            'detectedObjects': [],
            'isFireIncident': False,
            'analysis': '未检测到火灾或烟雾'
        }
```

```python
# box_drawer.py
def draw_detection_boxes(frame, detected_objects, class_colors):
    """在图像上绘制检测框"""

    result_frame = frame.copy()
    frame_height, frame_width = result_frame.shape[:2]

    for obj in detected_objects:
        # 1. 获取边界框和类别
        x1, y1, x2, y2 = [int(coord) for coord in obj['bbox']]
        class_name = obj['class']
        confidence = obj['confidence']

        # 2. 确保坐标在范围内
        x1 = max(0, min(x1, frame_width - 1))
        y1 = max(0, min(y1, frame_height - 1))
        x2 = max(0, min(x2, frame_width - 1))
        y2 = max(0, min(y2, frame_height - 1))

        # 3. 获取颜色
        color_hex = class_colors.get(class_name, '#FF0000')
        color_bgr = hex_to_bgr(color_hex)

        # 4. 绘制矩形框
        cv2.rectangle(
            result_frame,
            (x1, y1),
            (x2, y2),
            color_bgr,
            SERVER_BOX_LINE_THICKNESS
        )

        # 5. 绘制标签
        label_text = f"{class_name} {confidence * 100:.0f}%"

        # 计算文本大小
        font = cv2.FONT_HERSHEY_SIMPLEX
        text_size, _ = cv2.getTextSize(label_text, font, SERVER_FONT_SCALE, 1)

        # 绘制标签背景
        cv2.rectangle(
            result_frame,
            (x1, y1 - text_size[1] - 10),
            (x1 + text_size[0] + 10, y1),
            color_bgr,
            cv2.FILLED
        )

        # 绘制标签文本
        cv2.putText(
            result_frame,
            label_text,
            (x1 + 5, y1 - 5),
            font,
            SERVER_FONT_SCALE,
            (255, 255, 255),
            2
        )

    return result_frame
```

**绘制配置**:
- 线宽: 2像素
- 字体大小: 0.5
- 颜色: 火焰=红色 (#FF0000), 烟雾=橙色 (#FFA500)

```python
    """停止视频流并清理资源"""

    print(f"🛑 停止视频流: camera_id={camera_id}")

    stream = None

    # 1. 更新状态为stopping
    with stream_tasks_lock:
        if camera_id in stream_tasks:
            stream_tasks[camera_id]['status'] = 'stopping'

    # 2. 等待处理循环退出

    # 3. 获取stream对象并移除任务

        if camera_id in stream_tasks:
            stream_task = stream_tasks[camera_id]
            stream = stream_task.get('stream')
            stream_task['stream'] = None
            stream_tasks.pop(camera_id, None)

    # 4. 停止视频流
    if stream:
        try:
            stream.stop()
            print(f"✅ 视频流已停止: {camera_id}")
        except Exception as e:
            print(f"❌ 停止视频流异常: {e}")

    # 5. 更新ws_tasks
    with stream_tasks_lock:
        if ws_id in ws_tasks:
            ws_tasks[ws_id].discard(camera_id)

    # 6. 发送停止消息
    try:
        safe_ws_send(ws, ws_id, {
            'type': 'stream_stopped',
            'data': {
                'cameraId': camera_id
            }
        })
    except Exception as e:
        print(f"发送停止消息失败: {e}")

    # 7. 触发垃圾回收
    import gc
    gc.collect()
```

## 6. 关键技术实现

### 6.1 帧间隔控制

**配置文件**: `web-flask/config.py`

```python
FRAME_INTERVAL = 1.5  # 每1.5秒检测一次
```

**原因**:
- ✅ 降低CPU/GPU负载
- ✅ 减少WebSocket流量
- ✅ 平衡实时性和性能
- ✅ 火灾发展较慢，1.5秒足够

### 6.2 图像传输优化

#### 6.2.1 JPEG编码

```python
# 编码参数
_, buffer = cv2.imencode(
    '.jpg',                          # JPEG格式
    display_frame,
    [cv2.IMWRITE_JPEG_QUALITY, 85]  # 质量85%
)
```

**参数说明**:
- 格式: JPEG（有损压缩）
- 质量: 85%（平衡质量和大小）
- 平均大小: 50-150KB/帧

#### 6.2.2 Base64编码

```python
base64_frame = base64.b64encode(buffer).decode('utf-8')
result['frame'] = f"data:image/jpeg;base64,{base64_frame}"
```

**优势**:
- ✅ 直接嵌入WebSocket JSON消息
- ✅ 前端可直接用于`<img src>`
- ✅ 无需额外HTTP请求

**劣势**:
- ❌ 增加约33%数据量
- ❌ 编解码有计算开销

### 6.3 线程安全

#### 6.3.1 全局锁

```python
stream_tasks_lock = threading.Lock()

# 使用方式
with stream_tasks_lock:
    if camera_id in stream_tasks:
        stream_tasks[camera_id]['status'] = 'running'
```

**保护的共享数据**:
- `stream_tasks`: 所有任务字典
- `ws_tasks`: WebSocket任务集合

#### 6.3.2 WebSocket发送锁

```python
ws_send_locks: Dict[str, threading.Lock] = {}

def safe_ws_send(ws, ws_id, message):
    """线程安全发送"""
    with get_ws_send_lock(ws_id):
        ws.send(json.dumps(message))
```

**原因**: WebSocket不是线程安全的，并发发送会导致消息错乱

### 6.4 内存管理

#### 6.4.1 主动释放

```python
# 处理完帧后立即释放
del frame
if display_frame is not None:
    del display_frame
```

#### 6.4.2 垃圾回收

```python
import gc

# 处理结束后
gc.collect()

# GPU内存
if torch.cuda.is_available():
    torch.cuda.empty_cache()
```

#### 6.4.3 自动监控

```python
# 每30秒检查内存
memory_usage = process.memory_info().rss / 1024 / 1024

# 超过1GB触发GC
if memory_usage > 1000:
    gc.collect()
```

### 6.5 错误处理和重连

#### 6.5.1 RTSP连接重试

```python
def check_rtsp_connection(self) -> bool:
    """预检查RTSP可达性"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(2.0)  # 2秒超时

        result = sock.connect_ex((host, port))
        sock.close()

        return result == 0
    except Exception:
        return False
```

#### 6.5.2 WebSocket自动重连

```typescript
// 前端重连逻辑
function attemptReconnect() {

    ElMessage.error('无法连接到算法服务，请检查服务状态')
    return
  }

  reconnectAttempts++
  console.log(`尝试重连 (${reconnectAttempts}/${MAX_RECONNECT_ATTEMPTS})`)

  setTimeout(() => {
    connectWebSocket()
  }, 5000)  // 5秒后重连
}
```

## 9. 总结

### 9.1 系统特点

| 特点 | 说明 |
|------|------|
| **实时性** | WebSocket双向通信，端到端延迟2-3秒 |
| **智能性** | YOLO+LLM双引擎，误报率低 |
| **可靠性** | 自动重连、资源监控、异常恢复 |
| **可扩展性** | 支持多摄像头并发，线程隔离 |
| **性能可控** | 帧间隔控制、GPU加速、内存管理 |

### 9.2 技术亮点

1. **服务端AI处理**: 前端轻量化，计算集中管理
2. **WebSocket推送**: 无需轮询，实时性好
3. **双引擎检测**: YOLO定位 + LLM判断，准确率高
4. **资源自管理**: 自动监控清理，长期稳定运行
5. **并发支持**: 多线程架构，支持多路监控

### 9.3 适用场景

✅ **适合**:
- 中小规模监控（1-20路摄像头）
- 需要AI智能分析
- 对误报率要求高
- 有服务器计算资源

❌ **不适合**:
- 大规模监控（100+路）
- 低延迟要求（<500ms）
- 边缘计算场景
- 低带宽网络环境

### 9.4 改进方向

1. **批量处理**: 多个摄像头批量检测
2. **流媒体服务器**: 使用RTMP/HLS替代Base64传输
3. **边缘计算**: 部分检测移至摄像头端
4. **负载均衡**: 多台算法服务器分布式处理
5. **模型优化**: 模型量化、剪枝加速
